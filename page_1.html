RX62NのUSB機能を使う その2

2016-01-28 17:39:01 | RX62N
前回「RX62NのUSB機能を使う」でRX62NのUSB機能を利用してCDCデバイスでパソコンと通信するプログラムを紹介しましたが、パソコン側からうまく認識されないことがあるようです。ルネサスエレクトロニクス社のサンプルを見ていたら「Renesas USB MCU USB Peripheral Communication Device Class Driver(PCDC)」を見つけました。とりあえず手持ちのe2studioに読み込み実行しようとしましたが、E1デバッガ経由で実行することができませんでした。
プロジェクトを新たに作成し、ソースコードをコピーし、一部手直ししたら動作しました。

修正したファイル

demo_src\r_usb_pcdc_descriptor.c
  1. /******************************************************************************
  2. User define macro definitions
  3. ******************************************************************************/
  4. /* bcdUSB */
  5. #define USB_BCDNUM (0x0200u)
  6. /* Release Number */
  7. #define USB_RELEASE (0x0200u)
  8. /* DCP max packet size */
  9. #define USB_DCPMAXP (64u)
  10. /* Configuration number */
  11. #define USB_CONFIGNUM (1u)
  12. /* Vendor ID */
  13. #define USB_VENDORID 0x045B
  14. /* Product ID */
  15. #define USB_PRODUCTID 0x2014
 ベンダIDとプロダクトIDを手直しします。

r_config\r_bsp_config.h
使用しているMPUに合わせて変更(コメント通りに変更する)

「RXマイコン基板(RX62N) | Renesas MCU Software Libraryの仮想COMポートファームウェアを試す 」にも書かれています。

手直ししたプロジェクト

改造しやすいように手直ししました。
USB機能を動作させるためには、一定間隔で次のプログラムを実行する必要があります。
  1.         usb_cdc_driver(); /* USB Driver(Peripheral CDC) */
  2.         switch( cdc_dev_info.state ) /* Check application state */
  3.         {
  4.         case STATE_DATA_TRANSFER: /* Data transfer state */
  5.             lcd_update = cdc_data_transfer();
  6.         break;
  7.         case STATE_ATTACH: /* Wait Connect(USB Configured) state */
  8.             lcd_update = cdc_connect_wait();
  9.         break;
  10.         case STATE_DETACH:
  11.             cdc_detach_device(); /* Detach process */
  12.         break;
  14.         default:
  15.         break;
  16.         }
変更したプロジェクト

さらに、printf,scanfなどの標準入出力関数が使えるように変更しました。
一定間隔でusb_cdc_manage()関数を呼び出せば動作します。時間待ちの時にもこの関数を呼び出してください。

標準入出力が使えるプロジェクト

usb_cdc_manage()関数をタイマ割り込みで動かせば、プログラムが簡単にできると思い、プログラムを変更しました。タイマ割り込みで呼び出しても、問題なく動作しています。初期化のためinitUSB_cdc()関数を呼び出すだけで、パソコンとUSB通信することが可能です。

  8. void _INIT_IOLIB( void );    //プロトタイプ宣言
  10. int main(void)
  11. {
  12.     int data;
  13.     unsigned char c,buf[256];;
  15.     SYSTEM.SCKCR.BIT.ICK = 0x00;        //システムクロック　x16　96MHz
  16.     SYSTEM.SCKCR.BIT.PCK = 0x01;        //周辺クロック　　　x4 　48MHz
  17.     SYSTEM.SCKCR.BIT.BCK = 0x01;        //外部バスクロック　x4 　48MHz
  18.     /*Initialise the USB CDC Class*/
  19.     USBCDC_Init();

  34.     }
  35. }

RX62NでFreeRTOSを動かす

2015-11-27 22:05:41 | RX62N
RX62N(インターフェス2011年5月号付録)でFreeRTOSを動かしました。環境はe2 studio, renesas RXCコンパイラです。
ここにプロジェクトをエクスポートしたものを置いておきます。
ファイル構成は次の通りです。
”FreeRTOS”以下のフォルダに必要なファイルをまとめました。
”FreeRTOS/renesas”以下にウィザードが自動生成するファイルをまとめておいてあります。
”port.c”にエラーが表示されているのはインラインアセンブラを使用しているためです。



フォルダを閉じるとユーザープログラムのみでシンプルになります。



プロジェクト作成時にはできるだけ自動生成するファイルを省き、それでも作成されるものは”FreeRTOS/renesas”のファイルと被るので削除します。



ここに”FreeRTOS”フォルダ以下のファイルをドラッグアンドドロップでコピーすればFreeRTOSのインストールは終了です。



4つのタスクが違う間隔で起動と待期を繰り返しながらLEDを点滅させるプログラムは次のようになります。

  2. #include "FreeRTOS/renesas/iodefine.h"
  4. /* Kernel includes. */
  5. #include "FreeRTOS/FreeRTOS.h"
  6. #include "FreeRTOS/task.h"
  7. #include "FreeRTOS/queue.h"

  15.         vTaskDelay(100/portTICK_PERIOD_MS);
  16.     }
  17. }
  19. void vTask2(void *pvParameters)
  20. {

  24.     }
  25. }
  27. void vTask3(void *pvParameters)
  28. {
  29.     while(1) {
  30.         PORTD.DR.BIT.B3 = ~PORTD.DR.BIT.B3;
  31.         vTaskDelay(300/portTICK_PERIOD_MS);
  32.     }
  33. }
  35. void vTask4(void *pvParameters)
  36. {
  37.     while(1) {
  38.         PORTD.DR.BIT.B0 = ~PORTD.DR.BIT.B0;

  40.     }
  41. }
  43. void main(void)
  44. {

  48.     /* Turn all LEDs off. */
  49.     PORTD.DDR.BYTE = 0xFF;    //ポートC出力
  50.     PORTD.DR.BYTE = 0xFF;    //初期値
  53.     xTaskCreate(vTask1,"Task1",100,NULL,1,NULL);

RX62Nの12bitAD変換を使うその2

2015-11-26 15:00:48 | RX62N
RX62Nの12bitAD変換器を動かすプログラムのサンプルです。MTU0のTGRAのコンペアマッチを使い一定間隔で変換を行います。AD変換結果の読み出しに割り込みを使用しています。

  2. #include <stdio.h>
  3. #include <machine.h>
  4. #include "iodefine.h"
  5. #include "vect.h"
  7. #define PCLK    48    //PCLK = 48MHz
  9. /*
  10.  * MTU0の初期化
  11.  * TGRAのコンペアマッチでAD変換をスタートさせる
  12.  *
  13.  * 12bit A/Dコンバータ初期化
  14.  * AN0のみ使用
  15.  */
  16. void initMTU0_S12AD(void)
  17. {
  18.     MSTP(MTU0) = 0;                    //タイマMTU0回路動作
  19.     MTU0.TCR.BIT.CCLR = 1;            //TGRAコンペアマッチでカウンタクリア
  20.     MTU0.TCR.BIT.CKEG = 0;            //クロックエッジ選択ビット (使用しない)
  21.     MTU0.TCR.BIT.TPSC = 3;            //PCLK/64でカウント
  22.     MTU0.TMDR.BIT.MD = 0;            //通常動作
  23.     MTU0.TIER.BIT.TTGE = 1;            //A/D変換開始要求の発生を許可
  24.     MTU0.TCNT = 0;
  25.     MTU0.TGRA = PCLK*1000000L/64L/1000L - 1L;    // PCLK = 48MHzで1秒間に1000回AD変換を行う
  27.     MSTP(S12AD) = 0;                // 12bit　AD変換回路動作
  28.     delay_ms(100);                    // AD変換回路が安定するのを待つ
  29.     PORT4.DDR.BIT.B0 = 0;            // AN0を使用するためデジタル入力に
  30.     S12AD.ADCSR.BIT.CKS = 3;        // クロック　PCLK/1
  31.     S12AD.ADCSR.BIT.EXTRG = 0;        // A/D開始トリガ選択レジスタ（ADSTRGR）で選択されたタイマ要因による、スキャン変換の開始を選択
  32.     S12AD.ADCSR.BIT.TRGE = 1;        // 外部トリガ（ADTRG0#）または、MTU、TMRトリガによるスキャン変換を許可
  33.     S12AD.ADCSR.BIT.ADIE = 1;        // スキャン終了後のS12ADI0割り込み発生の許可
  34.     IR(S12AD,ADI) = 0;                // 割り込みフラグクリア
  35.     IEN(S12AD,ADI) = 1;                // 割込み許可
  36.     IPR(S12AD,ADI) = 0x01;            // 割り込み優先順位設定
  37.     S12AD.ADANS.BIT.ANS = 0x01;        // AN0のみ変換
  38.     S12AD.ADCER.BIT.ADRFMT = 0;        // 保存データは右詰め
  39.     S12AD.ADSTRGR.BIT.ADSTRS = 1;    // TRG0AN_0トリガで変換開始
  41.     MTUA.TSTR.BIT.CST0 = 1;            // タイマMTU0スタート
  42. }
  44. /*
  45.  * S12AD ADI
  46.  * AD変換終了で呼び出される割り込み関数
  47.  * AD変換の結果を表示
  48.  */
  49. //
  50. void Excep_S12AD_ADI(void)
  51. {
  52.     int ans;
  54.     ans = S12AD.ADDR0;
  55.     printf("AD value %d\n",ans);
  56. }
  59. int main(void)
  60. {
  61.     volatile int i;
  63.     Initialization();
  64.     initMTU0_S12AD();                        //AD変換初期化
  66.     printf("A/D converter test.\n");
  67.     while (1) {
  68.         //何もしない　割込みで動作
  69.     }
  71. }

RX62Nの12bitAD変換を使うその1

2015-11-26 14:51:43 | RX62N
RX62Nの12bitAD変換器を動かすプログラムのサンプルです。

  1. #include <stdio.h>
  2. #include <machine.h>
  3. #include "iodefine.h"
  5. /*
  6.  * 12bit A/Dコンバータ初期化
  7.  * AN0のみ使用
  8.  */
  9. void s12ad_init(void) {
  10.     MSTP(S12AD) = 0;
  11.     delay_ms(100);                // AD変換回路が安定するのを待つ
  12.     PORT4.DDR.BIT.B0 = 0;        // AN0を使用するためデジタル入力に
  13.     S12AD.ADCSR.BIT.CKS = 0x03;    // クロック　PCLK/1
  14.     S12AD.ADANS.BIT.ANS = 0x01;    // AN0のみ変換
  15.     S12AD.ADCER.BIT.ADRFMT = 0;    // 保存データは右詰め
  16. }
  18. /*
  19.  * AD変換を実行
  20.  */
  21. unsigned short get_s12ad(void) {
  23.   int i;
  24.   unsigned short adv;
  26.   S12AD.ADCSR.BIT.ADST = 1;         // AD変換開始
  27.   while(S12AD.ADCSR.BIT.ADST==1);    // AD変換が終了するのを待つ
  28.   adv = S12AD.ADDR0;                // 結果の読み出し
  30.   return adv;
  31. }
  33. int main(void)
  34. {
  35.     int adv;
  37.     Initialization();
  39.     s12ad_init();                        //AD変換初期化
  41.     printf("A/D converter test.\n");
  42.     while (1) {
  43.         adv = get_s12ad();
  44.         printf("AD value %d\n",adv);
  45.         delay_ms(100);
  46.     }
  48. }
